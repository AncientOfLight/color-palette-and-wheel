🎡 Color Wheel Component

A modular, interactive Color Wheel using HTML5 Canvas and vanilla JavaScript

Basic Implementation

The simplest way to initialize a Color Wheel:

<div id="color-wheel"></div>

<script src="color-wheel.js"></script>
<script>
  const wheel = new ColorWheel({
    container: '#color-wheel',
    size: 300
  });
</script>

This creates a 300x300px interactive color wheel that you can click and drag to select colors.

With Callbacks

Listen to color changes in real-time:

const wheel = new ColorWheel({
  container: '#color-wheel',
  size: 300,
  onColorChange: (color) => {
    console.log('HEX:', color.hex);
    console.log('RGB:', color.rgb);
    console.log('HSL:', color.hsl);
    console.log('HSV:', color.hsv);
  },
  onColorSelect: (color) => {
    console.log('Selected:', color);
  }
});

onColorChange fires during dragging, onColorSelect fires on mouse/touch release.

Output Formats

The Color Wheel returns colors in multiple formats:

HEX: #FF0000
RGB: rgb(255, 0, 0)
HSL: hsl(0, 100%, 50%)
HSV: {h, s, v}
const color = wheel.getColor();
// {
//   hex: '#FF0000',
//   rgb: { r: 255, g: 0, b: 0 },
//   hsl: { h: 0, s: 100, l: 50 },
//   hsv: { h: 0, s: 100, v: 100 }
// }

API Reference

new ColorWheel(options)

Creates a new Color Wheel instance.

Parameters:
options (Object) - Configuration object with properties listed above

Returns: ColorWheel instance 
const wheel = new ColorWheel({
  container: '#my-wheel',
  size: 250,
  initialColor: '#3B82F6',
  onColorChange: (color) => updatePreview(color)
});

setColor(hexColor)

Programmatically set the color on the wheel.

Parameters:
hexColor (String) - HEX color code (e.g., '#FF5733')

Returns: void 
wheel.setColor('#FF5733');

getColor()

Get the currently selected color in all formats.

Parameters: None

Returns: Object with properties: hex, rgb, hsl, hsv 
const selectedColor = wheel.getColor();
console.log(selectedColor.hex);    // '#FF5733'
console.log(selectedColor.rgb);    // { r: 255, g: 87, b: 51 }
console.log(selectedColor.hsl);    // { h: 10, s: 100, l: 60 }
console.log(selectedColor.hsv);    // { h: 10, s: 80, v: 100 }

destroy()

Clean up and remove the Color Wheel component from the DOM.

Parameters: None

Returns: void 
wheel.destroy(); // Removes canvas and listeners

Advanced Usage Examples

Example 1: Multiple Wheels

// Create a palette generator with multiple wheels
const wheels = [];
const colors = ['#FF0000', '#00FF00', '#0000FF'];

colors.forEach((color, index) => {
  const wheel = new ColorWheel({
    container: `#wheel-${index}`,
    size: 250,
    initialColor: color,
    onColorSelect: (selectedColor) => {
      savePaletteColor(index, selectedColor.hex);
    }
  });
  wheels.push(wheel);
});

// Later: Update all wheels
function updateAllWheels(newColors) {
  wheels.forEach((wheel, index) => {
    wheel.setColor(newColors[index]);
  });
}

Example 2: Synchronized Wheels

const primaryWheel = new ColorWheel({
  container: '#primary',
  size: 300,
  onColorChange: (color) => {
    // Update secondary wheel with same color
    secondaryWheel.setColor(color.hex);
  }
});

const secondaryWheel = new ColorWheel({
  container: '#secondary',
  size: 200
});

Example 3: Integration with Color Inputs

const wheel = new ColorWheel({
  container: '#color-wheel',
  size: 300,
  onColorChange: (color) => {
    // Update input fields
    document.getElementById('hex-input').value = color.hex;
    document.getElementById('rgb-input').value = 
      `rgb(${color.rgb.r}, ${color.rgb.g}, ${color.rgb.b})`;
    
    // Update preview
    document.getElementById('preview').style.background = color.hex;
  }
});

// Allow input to update wheel
document.getElementById('hex-input').addEventListener('change', (e) => {
  wheel.setColor(e.target.value);
});

Example 4: Real-time Theme Customizer

const wheel = new ColorWheel({
  container: '#theme-wheel',
  size: 300,
  onColorChange: (color) => {
    // Update CSS variables in real-time
    document.documentElement.style.setProperty(
      '--primary-color', 
      color.hex
    );
    
    // Update document.body background
    document.body.style.setProperty(
      '--primary-rgb',
      `${color.rgb.r}, ${color.rgb.g}, ${color.rgb.b}`
    );
  }
});

Performance Optimization Tips

// Tip 1: Debounce color changes for heavy operations
const wheel = new ColorWheel({
  container: '#wheel',
  onColorChange: debounce((color) => {
    performHeavyOperation(color);
  }, 100)
});

function debounce(func, delay) {
  let timeoutId;
  return function(...args) {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => func(...args), delay);
  };
}

// Tip 2: Use onColorSelect for final operations
// and onColorChange for visual feedback only
const wheel = new ColorWheel({
  container: '#wheel',
  onColorChange: (color) => {
    // Quick: Update preview instantly
    preview.style.background = color.hex;
  },
  onColorSelect: (color) => {
    // Heavy: Save to database after user finishes
    saveColorToDatabase(color);
  }
});

// Tip 3: Destroy wheels when no longer needed
function cleanupColorWheels() {
  wheels.forEach(wheel => wheel.destroy());
  wheels = [];
}
